/* Disable API Rest */
add_filter( 'rest_authentication_errors', function( $result ) {
    if ( ! empty( $result ) ) {
        return $result;
    }
    if ( ! is_user_logged_in() ) {
        return new WP_Error( 
        	'notPermision', 
        	'You are not currently logged in. By OCHOA Security',
        	array( 'status' => 401 ) 
        );
    }
    return $result;
});
